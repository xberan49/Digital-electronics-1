## 2. Seven-segment display decoder
1. * Listing of VHDL architecture from source file `hex_7seg.vhd`
2.  p_7seg_decoder : process(hex_i)
3.    begin
4.        case hex_i is
5.            when "0000" =>
6.                seg_o <= "0000001";     -- 0
7.            when "0001" =>
8.                seg_o <= "1001111";     -- 1
9.            when "0010" =>
10.               seg_o <= "0010010";     -- 2
11.           when "0011" =>
12.               seg_o <= "0000110";     -- 3
13.           when "0100" =>
14.               seg_o <= "1001100";     -- 4
15.           when "0101" =>
16.               seg_o <= "0100100";     -- 5
17.           when "0110" =>
18.               seg_o <= "0100000";     -- 6
19.           when "0111" =>
20.               seg_o <= "0001111";     -- 7
21.           when "1000" =>
22.               seg_o <= "0000000";     -- 8
23.           when "1001" =>
24.               seg_o <= "0000100";     -- 9
25.           when "1010" =>
26.               seg_o <= "0001000";     -- A
27.           when "1011" =>
28.               seg_o <= "1100000";     -- b
29.           when "1100" =>
30.               seg_o <= "0110001";     -- C
31.           when "1101" =>
32.               seg_o <= "1000010";     -- d
33.           when "1110" =>
34.               seg_o <= "0110000";     -- E
35.           when others =>
36.               seg_o <= "0111000";     -- F
37.       end case;
38.   end process p_7seg_decoder;
